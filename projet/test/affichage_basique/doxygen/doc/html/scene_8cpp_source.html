224 void SceneGL::paintGL()
225 {
226  if( ! _contenu_affichable)
227  return;
228 
229  // Gestion camera
230  glm::mat4 projection = glm::perspective(70.0f, float(640/480), 0.1f, 100.0f);
231  glm::mat4 modelview = glm::mat4(1.0);
232  glm::mat4 view = glm::lookAt(glm::vec3(_cam_pos.x(), _cam_pos.y(), _cam_pos.z()),
233  glm::vec3(0.0, 0.0, 0.0),
234  glm::vec3(0.0, 1.0, 0.0));
235 
236  _coef_rotation += 0.5;
237  modelview = glm::rotate(modelview, float(_coef_rotation*3.1415926535/180.0), glm::vec3(0.0, 1.0, 0.0));
238 
239  glm::mat4 mvp = projection * view * modelview;
240 
241  glEnable(GL_DEPTH_TEST);
242  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
243 
244  glUseProgram(_shader->get_id_shader_program());
245 
246  // Vecteur modifiant la teinture des couleurs des pixels
247  glUniform4fv(glGetUniformLocation(_shader->get_id_shader_program(), "Tint"), 1, &_teinture[0]);
248 
249  // Matrice de Modelview/View/Projection
250  glUniformMatrix4fv(glGetUniformLocation(_shader->get_id_shader_program(), "mvp"), 1, GL_FALSE, glm::value_ptr(mvp));
251 
252  glBindVertexArray(_sceneVAO);
253 
254  // On dessine les elements
255 #ifdef AFFDEBUG
256  std::cout << "affichage _sceneVAO[" << _sceneVAO
257  << "] ; Shad[" << _shader->get_id_shader_program()
258  << "] ; _sceneVBO[" << _sceneVBO
259  << "] ; _sceneEBO[" << _sceneEBO
260  << "] ; Pos[" << _position_loc << "]" << "|Col[" << _color_loc
261  << "]" << std::endl;
262 #endif
263 
264  glDrawElements(GL_LINES, _num_scene_indice, GL_UNSIGNED_INT, NULL);
265 }
